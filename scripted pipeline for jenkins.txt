node {
    try {
        stage('Build') {
            echo 'Building the project...'
            sh './build.sh'
        }

        stage('Test') {
            echo 'Running tests...'
            sh './test.sh'
        }

        stage('Deploy') {
            echo 'Deploying the project...'
            sh './deploy.sh'
        }
    } catch (Exception e) {
        currentBuild.result = 'FAILURE'
        throw e
    } finally {
        echo 'Cleaning up...'
    }
}
